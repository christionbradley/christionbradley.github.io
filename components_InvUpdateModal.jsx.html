import React, { useState } from 'react';

/**
 * Functional component representing a modal for updating the inventory by adding a new item.
 *
 * @component
 * @name InvUpdateModal
 * @param {Object} props - The properties passed to the component.
 * @param {boolean} props.inventoryUpdate - Flag indicating whether an update to the inventory data is needed.
 * @param {function} props.setInventoryUpdate - Function to set the flag for updating inventory data.
 * @param {function} props.onClose - Function to close the modal.
 * @returns {JSX.Element} The JSX element representing the inventory update modal.
 */
export const InvUpdateModal = (props) => {
    const [itemName, setItemName] = useState('');
    const [itemQuant, setItemQuant] = useState('');
    const [itemMin, setItemMin] = useState('');

    const setInventoryUpdateState = props.setInventoryUpdate;
    const inventoryUpdate = props.inventoryUpdate;
    const onClose = props.onClose;
const handleAddItemClick = () => {
    // logic to handle adding a menu item with itemName, itemPrice, and selectedItems
    var request = itemName + "+" + itemQuant + "+" + itemMin;
    request = request.replace(/ /g, "-");
    console.log(request);
    const addItem = async() => {
        try {
            const response = await fetch("https://project3-902-01-backend.onrender.com/addInvItem/" + request)
            if (response.ok) {
                setInventoryUpdateState(!inventoryUpdate);
            }
        }
        catch (error) {
            console.log("Error adding inventory item: ", error)
        }
    }
    addItem();
    onClose(); // Close the modal after handling the click
  };

  return (
    <>
      <div className="modal-overlay visible" onClick={onClose}></div>
      <div className="modal-container visible">
        <div className="modal-content visible">
          <h2>Add Inventory Item</h2>
          <div className="input-container">
            <label htmlFor="itemName">Item Name:</label>
            <input
              type="text"
              id="itemName"
              value={itemName}
              onChange={(e) => setItemName(e.target.value)}
            />
          </div>
          <div className="input-container">
            <label htmlFor="itemQuant">Item Quantity:</label>
            <input
              type="text"
              id="itemQuant"
              value={itemQuant}
              onChange={(e) => setItemQuant(e.target.value)}
            />
          </div>
          <div className="input-container">
            <label htmlFor="itemMin">Item Minimum:</label>
            <input
              type="text"
              id="itemMin"
              value={itemMin}
              onChange={(e) => setItemMin(e.target.value)}
            />
          </div>
          <button onClick={handleAddItemClick}>Add Item</button>
          <button onClick={onClose}>Close</button>
        </div>
      </div>
    </>
  );
};
