import React, { useState, useEffect } from 'react';
import { MenuProduct } from '../../components/menuProduct';
// import "../../MotherCSS.css";

/**
 * Functional component representing the menu page.
 *
 * @component
 * @name Menu
 * @returns {JSX.Element} The JSX element representing the menu page.
 */
export const Menu = () => {
    const [products, setProducts] = useState({});

    useEffect(() => {
         /**
         * Fetches products from the backend and sets them to state.
         *
         * @async
         * @function
         * @name fetchProducts
         * @memberof Menu
         * @inner
         */
        const fetchProducts = async () => {
            try {
                const response = await fetch('https://project3-902-01-backend.onrender.com/currentMenu');
                // const response = await fetch('http://127.0.0.1:5000/currentMenu');
                if (response.ok) {
                    const data = await response.json();
                    setProducts(data); // Set the fetched products to state
                } else {
                    // Handle the error if the fetch fails
                    console.error('Failed to fetch products');
                }
            } catch (error) {
                console.error('Error fetching products:', error);
            }
        };

        fetchProducts();
    }, []);


    return (
        <div className='menu-root'>
            <div className='shop-container'>
                <div className='shop'>
                    <div className='shopTitle'>
                        <h1>Menu</h1>
                    </div>
                    <div className='products'>
                        {Object.values(products).map((p) => (
                            <MenuProduct 
                                id={p.id}
                                productName={p.item}
                                price={p.price}
                                productImage={p.imglink}
                                description={p.description}
                            />
                        ))}
                    </div>
                </div>
            </div>

        </div>
    )
}
