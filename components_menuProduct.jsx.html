import React from 'react';

/**
 * Functional component representing an individual product in the menu.
 *
 * @component
 * @name MenuProduct
 * @param {Object} props - The properties passed to the component.
 * @param {string} props.id - The unique identifier for the product.
 * @param {string} props.productName - The name of the product.
 * @param {string} props.price - The price of the product.
 * @param {string} props.productImage - The URL or path to the image of the product.
 * @param {string} props.description - The description or details of the product.
 * @param {function} props.updateCart - Function to handle adding the product to the cart.
 * @returns {JSX.Element} The JSX element representing an individual product.
 */
export const MenuProduct = (props) => {
    const id = props.id;
    const productName = props.productName;
    const price = props.price;
    const productImage = props.productImage;
    const description = props.description;

    const addToCartHandler = (product) => {
        props.updateCart(product);
    }

    return (
        <div className='product'>
            <img src={productImage} alt={productName} />
            <div className='description'>
                <p className='product-name'><b>{productName}</b></p>
                <p className='product-price'><b>{price}</b></p>
                <p className='product-description'>{description}</p>
            </div>
            
            {/* <button className='add-to-cart-button' onClick={() => addToCartHandler(productName)}>Add To Cart</button> */}
        </div>
    )
}
