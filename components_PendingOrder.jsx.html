import React from 'react'

/**
 * PendingOrder component for displaying and managing pending orders.
 * @component
 * @param {Object} props - The component props.
 * @param {number} props.id - The order ID.
 * @param {Array.<string>} props.items - The array of items in the order.
 * @param {boolean} props.pendingToggle - The pending toggle status.
 * @param {Function} props.setPendingToggle - Function to set the pending toggle status.
 * @returns {JSX.Element} The PendingOrder component.
 */
export const PendingOrder = ({id, items, pendingToggle, setPendingToggle}) => {

    const handleCancel = () => {
        const cancel = async() => {
            try{
                const response = await fetch('https://project3-902-01-backend.onrender.com/cancelOrder/'+id)
                if(response.ok){
                    console.log('canceled order')
                }
                else{console.error("couldn't complete order")}
            } catch(error){console.error('error complete products', error)}
            setPendingToggle(!pendingToggle);
        }
        cancel();
    }

    const handleComplete = (id) => {
        const complete = async() => {
            try{
                const response = await fetch('https://project3-902-01-backend.onrender.com/completeOrder/'+id)
                if(response.ok){
                    console.log('completed order')
                }
                else{console.error("couldn't complete order")}
            } catch(error){console.error('error complete products', error)}
            setPendingToggle(!pendingToggle);
        }
        complete();
    }


    return (
        <div className='pending-product'>
            <div className='description'>
                <p className='product-name'><b>Order number:</b> {id}</p>
                <p className='product-price'><b>Items:</b> {items.join(', ')}</p>
            </div>
            <button onClick={()=>{handleComplete(id)}}>Complete</button>
            <button onClick={()=>{handleCancel(id)}}>Cancel</button>
        </div>
    )
}
