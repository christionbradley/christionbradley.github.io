import React, { useEffect, useState } from 'react';

/**
 * Functional component representing a modal for adding a new employee.
 *
 * @component
 * @name AddEmployeeModal
 * @param {Object} props - The properties passed to the component.
 * @param {boolean} props.visible - Indicates whether the modal is visible or not.
 * @param {function} props.onClose - Function to close the modal.
 * @param {boolean} props.employeeUpdate - State variable for employee update trigger.
 * @param {function} props.setEmployeeUpdate - Function to set the employee update state.
 * @returns {JSX.Element} The JSX element representing the modal for adding a new employee.
 */
export const AddEmployeeModal = (props) => {
  const [newName, setNewName] = useState('');
  const [newPosition, setNewPosition] = useState(''); // Changed to a default value, can be changed based on your needs
  const [error, setError] = useState('');

  //   const visible = props.visible;
  const onClose = props.onClose;
  const employeeUpdate = props.employeeUpdate;
  const setEmployeeUpdate = props.setEmployeeUpdate;

  // Handles the click event when the "Add Employee" button is clicked.
/**
 * Handles the click event when the "Add Employee" button is clicked.
 * Validates input, formats the request, and sends a request to add a new employee.
 * Closes the modal after the request is processed.
 *
 * @function
 * @name handleAddItemClick
 */
  const handleAddItemClick = () => {
    if (newName.trim() === '' || newPosition === '') {
        setError('Both name and position are required.');
        return;
    }
    let newName2;
    newName2 = newName;
    newName2 = newName2.replace(/ /g, "-");
    var request = newName2 + "&" + newPosition;

    /**
     * Sends a request to add a new employee to the backend.
     *
     * @async
     * @function
     * @name addNewEmployee
     * @param {string} req - The formatted request string.
     */
    const addNewEmployee = async(req) => {
      try{
        const response = await fetch('https://project3-902-01-backend.onrender.com/createEmployee/'+req)
        if(response.ok){
          setEmployeeUpdate(!employeeUpdate);
        } else {console.error("failed to add new employee");}
      } catch(error){console.error("failed to connect to backend");}
    }

    addNewEmployee(request);
    console.log(request);
    onClose();
  };

  const positionOptions = ['Admin', 'Manager', 'Cashier'];

  return (
    <>
      <div className="modal-overlay visible" onClick={onClose}></div>
      <div className="modal-container visible">
        <div className="modal-content visible">
          <h2>Add New Employee</h2>
          <div className="input-container">
            <label htmlFor="newName">Employee Name:</label>
            <input
              type="text"
              id="newName"
              value={newName}
              onChange={(e) => setNewName(e.target.value)}
            />
          </div>
          <div className="input-container">
            <label htmlFor="newPosition">Position:</label>
            <select
              id="newPosition"
              value={newPosition}
              onChange={(e) => setNewPosition(e.target.value)}
            >
              <option value="" disabled>Select position</option>
              {positionOptions.map((position) => (
                <option key={position} value={position}>
                  {position}
                </option>
              ))}
            </select>
          </div>
          <button onClick={handleAddItemClick}>Add Employee</button>
          <button onClick={onClose}>Close</button>
        </div>
      </div>
    </>
  );
};
