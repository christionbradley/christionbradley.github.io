import React, { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';
import { ArrowCircleRight } from 'phosphor-react';

/**
 * Functional component representing a cashier authentication modal.
 *
 * @component
 * @name CashierModal
 * @param {Object} props - The properties passed to the component.
 * @param {function} props.onClose - Function to close the modal.
 * @param {number} props.currLogged - Current logged-in state.
 * @param {function} props.setCurrLogged - Function to set the current logged-in state.
 * @returns {JSX.Element} The JSX element representing the cashier authentication modal.
 */
export const CashierModal = ({ onClose, currLogged, setCurrLogged }) => {
    const [inputValue, setInputValue] = useState('');
    const specificString = '9876';

    // Use a separate state to handle the condition
    const [isAdmin, setIsAdmin] = useState(false);

    useEffect(() => {
        // Update the state when the condition is met
        if (inputValue === specificString) {
            setIsAdmin(true);
        } else {
            setIsAdmin(false);
        }
    }, [inputValue]);

    const handleLinkClick = () => {
        // Update the parent component state outside the render phase
        if (isAdmin) {
            setCurrLogged(1);
        }
    };

    return (
        <div className='cashier-modal'>
            <div className='cashier-overlay'></div>
            <div className='cashier-modal-content'>
                <div className='input-div'>
                    <p>Input Cashier Password (9876)</p>
                    <input
                        value={inputValue}
                        onChange={(e) => setInputValue(e.target.value)}
                    />
                </div>
                {isAdmin && (
                    <Link to="/cashier" onClick={handleLinkClick}>
                        <ArrowCircleRight size={32}/>
                    </Link>
                )}
                <div className='button-container'>
                    <button className='exit' onClick={onClose}>
                        Cancel
                    </button>
                </div>
            </div>
        </div>
    );
};
