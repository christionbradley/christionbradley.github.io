import React from 'react'

/**
 * Product component for displaying product information.
 * @component
 * @param {Object} props - The component props.
 * @param {number} props.id - The product ID.
 * @param {string} props.productName - The name of the product.
 * @param {number} props.price - The price of the product.
 * @param {string} props.productImage - The URL of the product image.
 * @param {Function} props.updateCart - Function to update the shopping cart.
 * @param {Function} props.updateCost - Function to update the total cost.
 * @param {Function} props.updateCartPrices - Function to update cart prices.
 * @returns {JSX.Element} The Product component.
 */
export const Product = (props) => {
    const id = props.id;
    const productName = props.productName;
    const price = props.price;
    const productImage = props.productImage;
    // const { id, productName, price, productImage } = props.data;

    const addToCartHandler = (product) => {
        props.updateCart(product);
    }

    const updateCostHandler = (price) => {
        props.updateCost(price);
    }

    const updateCartPriceHandler = (newPrice) => {
        props.updateCartPrices(newPrice);
    }

    const clickHandler = () => {
        addToCartHandler(productName);
        updateCostHandler(price);
        updateCartPriceHandler(price);
    }

    return (
        <div className='product' onClick={clickHandler}>
            <img src={productImage} />
            <div className='description'>
                <p className='product-name'><b>{productName}</b></p>
                <p className='product-price'><b>{price}</b></p>
            </div>
        </div>
    )
}
