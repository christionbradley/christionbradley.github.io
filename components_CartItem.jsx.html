import React from 'react';

/**
 * Functional component representing an item in the shopping cart.
 *
 * @component
 * @name CartItem
 * @param {Object} props - The properties passed to the component.
 * @param {number} props.id - The unique identifier of the cart item.
 * @param {string} props.currItem - The name of the current item.
 * @param {Array} props.currCart - The current state of the shopping cart.
 * @param {function} props.removeFromCart - Function to remove an item from the cart.
 * @param {function} props.removeFromCartPrices - Function to update the cart prices after removing an item.
 * @param {function} props.reloadCart - Function to reload the shopping cart.
 * @returns {JSX.Element} The JSX element representing an item in the shopping cart.
 */
export const CartItem = ({ id, currItem, currCart, removeFromCart, removeFromCartPrices, reloadCart }) => {
  const handleClick = () => {
    // console.log("Before: " + currCart);
    removeFromCart(id-1);
    removeFromCartPrices(id-1);
    reloadCart();
    // console.log("After: " + currCart);
    
  };

  return (
    <div className="cart-item">
      <p className="item-number">{id}</p>
      <p className="item-name">{currItem}</p>
      <button className='remove-cart-item' onClick={handleClick}>X</button>
    </div>
  );
};
