import React, { useState } from 'react';

/**
 * Functional component representing a date range entry modal.
 *
 * @component
 * @name DateRangeEntryModal
 * @param {Object} props - The properties passed to the component.
 * @param {boolean} props.isOpen - Flag indicating whether the modal is open or closed.
 * @param {function} props.onClose - Function to close the modal.
 * @param {function} props.onSubmit - Function to handle the submission of the selected date range.
 * @returns {JSX.Element} The JSX element representing the date range entry modal.
 */
export const DateRangeEntryModal = ({ isOpen, onClose, onSubmit }) => {
  const [startDay, setStartDay] = useState('');
  const [startMonth, setStartMonth] = useState('');
  const [startYear, setStartYear] = useState('');

  const [endDay, setEndDay] = useState('');
  const [endMonth, setEndMonth] = useState('');
  const [endYear, setEndYear] = useState('');

  const handleStartDateChange = (e) => {
    const { name, value } = e.target;
    if (name === 'day') setStartDay(value);
    else if (name === 'month') setStartMonth(value);
    else if (name === 'year') setStartYear(value);
  };

  const handleEndDateChange = (e) => {
    const { name, value } = e.target;
    if (name === 'day') setEndDay(value);
    else if (name === 'month') setEndMonth(value);
    else if (name === 'year') setEndYear(value);
  };

  const handleSubmit = () => {
    const startDate = `${startYear}-${startMonth}-${startDay}`;
    const endDate = `${endYear}-${endMonth}-${endDay}`;
    onSubmit({ startDate, endDate });
    setStartDay('');
    setStartMonth('');
    setStartYear('');
    setEndDay('');
    setEndMonth('');
    setEndYear('');
    onClose();
  };

  return (
    <div style={{ display: isOpen ? 'block' : 'none', position: 'absolute', top: '50%', left: '50%', transform: 'translate(-50%, -50%)', padding: '20px', background: '#fff', border: '1px solid #ccc', borderRadius: '8px', zIndex: '1000' }}>
      <div>
        <label>
          Start Date:
          <select name="day" value={startDay} onChange={handleStartDateChange}>
            <option value="">--</option>
            {[...Array(31).keys()].map((day) => (
              <option key={day + 1} value={(day + 1).toString().padStart(2, '0')}>{day + 1}</option>
            ))}
          </select>
          <select name="month" value={startMonth} onChange={handleStartDateChange}>
            <option value="">--</option>
            {[...Array(12).keys()].map((month) => (
              <option key={month + 1} value={(month + 1).toString().padStart(2, '0')}>{month + 1}</option>
            ))}
          </select>
          <select name="year" value={startYear} onChange={handleStartDateChange}>
            <option value="">--</option>
            {[...Array(10).keys()].map((year) => (
              <option key={2023 - year} value={2023 - year}>{2023 - year}</option>
            ))}
          </select>
        </label>
      </div>
      <div>
        <label>
          End Date:
          <select name="day" value={endDay} onChange={handleEndDateChange}>
            <option value="">--</option>
            {[...Array(31).keys()].map((day) => (
              <option key={day + 1} value={(day + 1).toString().padStart(2, '0')}>{day + 1}</option>
            ))}
          </select>
          <select name="month" value={endMonth} onChange={handleEndDateChange}>
            <option value="">--</option>
            {[...Array(12).keys()].map((month) => (
              <option key={month + 1} value={(month + 1).toString().padStart(2, '0')}>{month + 1}</option>
            ))}
          </select>
          <select name="year" value={endYear} onChange={handleEndDateChange}>
            <option value="">--</option>
            {[...Array(10).keys()].map((year) => (
              <option key={2023 - year} value={2023 - year}>{2023 - year}</option>
            ))}
          </select>
        </label>
      </div>
      <button onClick={handleSubmit}>Submit</button>
      <button onClick={onClose}>Cancel</button>
    </div>
  );
};
